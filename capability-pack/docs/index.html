Canonical flow (copy)

export RPC_URL="https://sepolia.base.org"
export CONTRACT="0x..."
export PRIVATE_KEY="0x..."

PRICE_WEI=$(cast call --rpc-url "$RPC_URL" "$CONTRACT" "getPrice(string,uint8)(uint256)" "alice" 1 | awk '{print $1}')
cast send --rpc-url "$RPC_URL" --private-key "$PRIVATE_KEY" "$CONTRACT" "register(string,uint8)" "alice" 1 --value "$PRICE_WEI"
cast send --rpc-url "$RPC_URL" --private-key "$PRIVATE_KEY" "$CONTRACT" "setCID(string,string)" "alice" "bafy..."

curl -i https://alice.ipns.io/
